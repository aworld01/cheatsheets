Queries

CREATE TABLE Product(Id INTEGER PRIMARY KEY AUTOINCREMENT, Name TEXT NOT NULL, Price REAL, Quantity INTEGER);
to create a table.
INSERT INTO Product(Name, Price, Quantity) VALUES('AutoCAD', 200, 20);
to insert data.
SELECT * FROM Product;
to see all table records.
SELECT Name FROM Product;
to see name of all records.
SELECT Name, Quantity FROM Product;
to see name and quantity of all records.
UPDATE Product SET Name = "Photoshop" WHERE Name = "MS Office";
to update record.
DELETE FROM Product WHERE Id = 1;
to delete record.
DROP TABLE Product;
to delete the table.
