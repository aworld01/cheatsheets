There are two options to change permision settings

Option-1 
        USE mysql;
        SELECT User, Host, Plugin from mysql.user;
        UPDATE User SET Plugin='mysql_native_password' WHERE User='root';
        FLUSH PRIVILEGES;
        exit
        service mysql restart
Option-2 
        USE mysql;
        CREATE USER 'your_system_user'@'localhost' IDENTIFIED VIA mysql_native_password;
        GRANT ALL PRIVILEGES ON *.* TO 'your_system_user'@'localhost';
        UPDATE User SET PLUGIN 'auth_socket' WHERE User='your_system_user';
        FLUSH PRIVILEGES;
        exit
        service mysql restart

service mysql start: to start mysql service.

service mysql stop: to stop mysql service.

service mysql restart: to restart mysql service.

service mysql status: to see status.

mysql: to start mysql.

mysql -u thor: to start mysql with thor user.

mysql -u thor -p: to start mysql with thor user and password.

SHOW VARIABLES WHERE Variable_Name = 'port'; to see port number.

Change password

--for MySQL 
        ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'root';
--for MariaDB 
        ALTER USER 'root'@'localhost' IDENTIFIED VIA mysql_native_password USING PASSWORD('root');
