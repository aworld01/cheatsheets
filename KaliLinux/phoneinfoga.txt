sudo curl -sSL https://raw.githubusercontent.com/sundowndev/phoneinfoga/master/support/scripts/install | bash

You can now install the program globally : sudo mv ./phoneinfoga /usr/bin/phoneinfoga

sudo ./phoneinfoga serve -p 8080: to scan on webserver
