#!/bin/bash num=0 until [ $num -gt 10 ] do echo $num num=$((num+1)) done

#!/bin/bash num=0 until [ $num -gt 10 ] do echo $num ((num++)) done

#!/bin/bash num=0 until [ $num -gt 10 ] do echo $num ((++num)) done

#!/bin/bash num=0 until (( $num > 10 )) do echo $num ((++num)) done
