There are two types of variable in linux os:-

System variable
User defined variable

System variable

#!/bin/bash echo This is my shell location: $BASH echo This is my bash version: $BASH_VERSION echo My Present Working Directory location is: $PWD

User defined variable

#!/bin/bash name='Abdul Zoha' echo My name is $name

to take user input.

#!/bin/bash clear read -p "Enter your name: " name echo "Hi $name ! How are you today?"

#!/bin/bash clear read -p "Enter your name: " name read -sp "Enter your password: " password echo echo "Hi $name ! How are you today?" echo "Your password is: $password"

#!/bin/bash clear echo "Enter the list of the names: " read -a names echo "You have entered: ${names[0]}, ${names[1]}, ${names[2]}"

arithmetic operators

# syntax # + = to addition # - = to substraction # * = to multiplication # / = to division. # % = to modulas # ** = to exponent

to addition.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 add=$(($num1 + $num2)) echo "Addition is: $add"

to substraction.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 subs=$(($num1 - $num2)) echo "Substraction is: $subs"

to multiplication.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 mult=$(($num1 * $num2)) echo "Substraction is: $mult"

to division.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 div=$(($num1 / $num2)) echo "Division is: $div"

to modulas.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 a=$(($num1 % $num2)) echo "Remander is: $a"

to exponent.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter any two numbers separated by space: " num1 num2 e=$(($num1 ** $num2)) echo "Exponent is: $e"

to addition by two input.

#!/bin/bash clear echo -e "\t\t\tLearning Arithmetic operator:-" read -p "Enter the first number: " num1 read -p "Enter the second number: " num2 e=$(($num1 + $num2)) echo "Result: $e"
