    With block/Context manager

    ''' The with statement can be used while openinig a file. When we open a file using with statement there is no need to close the file explicitly. '''
  1. with open("file.txt", mode="r", encoding="utf-8") as f: #read mode. data = f.read() #to read a file. print(data)
  2. with open("file.txt", mode="w", encoding="utf-8") as f: #write mode. f.write("Hello world!") #to write a file.
  3. with open("file.txt", mode="a", encoding="utf-8") as f: #append mode. f.write("I'm Abdul Zoha") #to write new data next to old data.
  4. with open("file.txt", mode="r+", encoding="utf-8") as f: #read + write mode. data = f.read() #to read file print(data) f.write("I'm Abdul Zoha") #to append file f.seek(0) #to seek cursor at 0 index data2 = f.read() #to see updated file print(data2)
  5. with open("file.txt", mode="r+", encoding="utf-8") as f: #read + write mode. data = f.read() l = len(data) #to move cursor at last of data f.seek(l) f.write("Artificialworld01") #to write new data next to old data.

    6. Write one file data to another file.

  6. with open("file.txt", mode="r", encoding="utf-8") as rf: with open("file2.txt", mode="w", encoding="utf-8") as wf: wf.write(rf.read())
