Making CSS snippets

    // example-1
        "flex shortcut": {
            "prefix": "dflex",
         	"body": "display: flex;",
         	"description": "to center the div using flex"
         }
    
    // example-2
        "flex shortcut": {
        	"prefix": "dflex",
        	"body": [
        		"display: flex;",
        		"justify-content: center;",
        		"align-items: center;"
        	],
            "description": "to center the div using flex"
        }
    
    // example-3
        "width height shortcut": {
            "prefix": "wh",
            "body": [
                "width: $1;",
                "height: $2;"
            ],
            "description": "width and height"
        }
