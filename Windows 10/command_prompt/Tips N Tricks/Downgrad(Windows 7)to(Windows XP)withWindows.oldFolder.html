Date: 02/05/2019

Downgrad (Windows 7) to (Windows XP) with Windows.old folder

  1. Boot your pc with (Windows 7) CD.
  2. When it ask to Install or Repair press
    Shift
    +
    F10
    (CMD Shuld be appare)

    3. Then type the command step by step below

    1. x:\>
      c:
      ENTER

    2. c:\>
      md W7
      ENTER
    3. c:\>
      move Windows W7\Windows
      ENTER
    4. c:\>
      move "Program Files" "W7\Program Files"
      ENTER
    5. c:\>
      move Users W7\Users
      ENTER
    6. c:\>
      attrib -h -s -r ProgramData
      ENTER
    7. c:\>
      move ProgramData W7\ProgramData
      ENTER
    8. c:\>
      rd "Documents and Settings"
      ENTER
    9. c:\>
      move /y c:\Windows.old\Windows c:\
      ENTER
    10. c:\>
      move /y "c:\Windows.old\Program Files" c:\
      ENTER
    11. c:\>
      move /y "c:\Windows.old\Documents and Settings"
      ENTER
    12. c:\>
      e:\boot\bootsect /nt52 c:
      ENTER
    13. c:\>
      attrib -h -s -r boot.ini.saved
      ENTER
    14. c:\>
      copy boot.ini.saved boot.ini
      ENTER
    15. c:\>
      exit
      ENTER
